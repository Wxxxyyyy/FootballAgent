> **目标**: 1-2 天内跑通一个可用的世界杯预测原型，复用现有模型和 LLM，OpenClaw 容器内写 loop 动态抓取赔率。

> **原则**: 快速可用优先，暂不纠结工程化。文件存储、单进程、够用即可。世界杯已踢一半，时间紧迫。

### Step 2: 世界杯预测脚本（Day 1 下午）

> 目标：单文件脚本，从 OpenClaw 拿赔率 → ML 预测 → LLM 分析 → 存 JSON

新建 `agents/worldcup_agent/quick_predict.py`：

```python
"""
世界杯快速预测脚本
用法: python -m agents.worldcup_agent.quick_predict
"""
import httpx, json, os, time
from datetime import datetime
from pathlib import Path

OPENCLAW_URL = os.getenv("OPENCLAW_API_URL", "http://127.0.0.1:9000")
OUTPUT_DIR = Path("data/worldcup/predictions")
OUTPUT_DIR.mkdir(parents=True, exist_ok=True)

def fetch_odds():
    """从 OpenClaw 拉取当前世界杯赔率"""
    resp = httpx.post(f"{OPENCLAW_URL}/task", json={"task_type": "fetch_worldcup_odds"})
    return resp.json().get("result", {})

def predict_match(match):
    """单场预测：复用现有 ML + LLM"""
    from agents.predicted_agent.models.statistical_model import OddsModel
    from agents.predicted_agent.models.llm_predictor import predict_with_llm

    model = OddsModel.load()
    odds = match["odds"]  # B365H, B365D, B365A, Over25, Under25, AHh

    # ML 预测（复用现有模型）
    ml_result = model.predict_from_odds(
        b365h=odds["B365H"], b365d=odds["B365D"], b365a=odds["B365A"],
        b365_over25=odds["Over25"], b365_under25=odds["Under25"], ahh=odds.get("AHh", 0),
    )

    # LLM 分析（复用现有 llm_predictor，传入世界杯上下文）
    llm_result = predict_with_llm(
        home_team=match["home"], away_team=match["away"],
        date=match.get("date", ""), ml_result=ml_result,
        home_last_5=[], away_last_5=[], h2h_records=[],
        odds_info=odds, upset_signals=None,
    )

    return {"ml": ml_result, "llm": llm_result}

def run():
    odds_data = fetch_odds()
    for match in odds_data.get("matches", []):
        if match.get("status") != "upcoming":  # 只预测未开赛
            continue
        result = predict_match(match)
        result["prediction_time"] = datetime.now().isoformat()
        result["odds_snapshot"] = match["odds"]

        filename = f"{match['date']}_{match['home']}_vs_{match['away']}.json"
        with open(OUTPUT_DIR / filename, "w", encoding="utf-8") as f:
            json.dump(result, f, ensure_ascii=False, indent=2)
        print(f"[预测] {match['home']} vs {match['away']} → {result['llm'].get('wdl_prediction', {})}")

if __name__ == "__main__":
    run()
```

**关键复用点**：
- `OddsModel.load()` 直接用五大联赛训练的模型（世界杯数据太少，不值得单独训练）
- `predict_with_llm()` 直接复用，在 prompt 里补充世界杯上下文即可
- 不碰 MySQL/Neo4j/Redis，纯文件操作

### Step 4: 赔率变动触发重新预测（Day 1 晚上）

> 目标：OpenClaw loop 检测到赔率大幅变动时，通知预测端重新跑

在 OpenClaw 容器的 `app.py` 里加变动检测：

```python
# app.py odds_loop() 内部
def odds_loop():
    prev_odds = {}
    while True:
        try:
            # ... 抓取新赔码 ...
            new_odds = _parse_odds(matches)

            # 检测变动
            for match_id, odds in new_odds.items():
                prev = prev_odds.get(match_id, {})
                if prev:
                    shift = abs(odds["B365H"] - prev.get("B365H", odds["B365H"])) / prev.get("B365H", odds["B365H"])
                    if shift > 0.10:  # 赔率变动 >10%
                        print(f"[变动] {match_id} 主胜赔率变动 {shift:.1%}")
                        # 触发重新预测（调本机预测脚本 或 标记待重测）
                        _trigger_repredict(match_id)

            prev_odds = new_odds
        except Exception as e:
            print(f"[Loop] 失败: {e}")
        time.sleep(900)
```

### Step 5: 命中率统计脚本（Day 2）

> 目标：比赛结束后录入实际结果，统计命中率

新建 `agents/worldcup_agent/track_accuracy.py`：

```python
"""
世界杯预测命中率统计
用法:
  1. 比赛结束后手动录入实际结果
  2. python -m agents.worldcup_agent.track_accuracy
"""
import json, glob
from pathlib import Path

PRED_DIR = Path("data/worldcup/predictions")

def update_actual(match_file, home_goals, away_goals):
    """录入实际比分"""
    data = json.loads(Path(match_file).read_text(encoding="utf-8"))
    data["actual_result"] = {"score": f"{home_goals}:{away_goals}", "wdl": "H" if home_goals>away_goals else ("D" if home_goals==away_goals else "A")}
    # 计算命中
    pred_wdl = data["llm"]["wdl_prediction"]["primary"]
    data["hit"] = {"wdl": pred_wdl == data["actual_result"]["wdl"]}
    Path(match_file).write_text(json.dumps(data, ensure_ascii=False, indent=2), encoding="utf-8")

def report():
    """统计命中率"""
    files = glob.glob(str(PRED_DIR / "*.json"))
    wdl_hits = sum(1 for f in files if json.loads(Path(f).read_text())["hit"]["wdl"])
    print(f"总预测: {len(files)}, WDL 命中: {wdl_hits}, 命中率: {wdl_hits/len(files):.1%}")
```
